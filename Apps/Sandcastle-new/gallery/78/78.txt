var viewer = new Cesium.Viewer('cesiumContainer', {
    infoBox : false,
    selectionIndicator : false
});

function addCZML(id){
    viewer.dataSources.removeAll(false);
    var url = "../SampleData/" + id + ".czml";
    Cesium.CzmlDataSource.load(url).then(function(val){
        var ds = val;
        viewer.dataSources.add(ds);
        var modelEntity = viewer.dataSources.get(0).entities.getById(id);
        viewer.trackedEntity = modelEntity;
    });
}


var options = [{
    text : 'Aircraft',
    onselect : function() {
        addCZML('3dmodels_aircraft');
    }
}, {
    text : 'Ground vehicle',
    onselect : function() {
        addCZML('3dmodels_groundvehicle');
    }
}, {
    text : 'Milk truck',
    onselect : function() {
        addCZML('3dmodels_milktruck');
    }
}, {
    text : 'Skinned character',
    onselect : function() {
        addCZML('3dmodels_skinnedcharacter');
    }
}];

Sandcastle.addToolbarMenu(options);
